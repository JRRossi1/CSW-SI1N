import pandas as pd

            def remove_outlier(df_old):
                # Cálculo do primeiro e terceiro quartil
                Q1 = df_old.quantile(0.25)
                Q3 = df_old.quantile(0.75)
                IQR = Q3 - Q1
                
                # Definindo limites para identificar outliers
                lower_bound = Q1 - 1.5 * IQR
                upper_bound = Q3 + 1.5 * IQR
                
                # Filtrando o DataFrame para remover outliers
                df_new = df_old[~((df_old < lower_bound) | (df_old > upper_bound)).any(axis=1)]
                
                return df_new
